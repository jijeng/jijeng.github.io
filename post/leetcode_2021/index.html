class Solution:
    # m 表示重复的 subarry，k 表示重复的次数
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        
        n =len(arr)
        counter =0
        # 按顺序遍历保证了 consecutively 
        for i in range(n-m):
            
            if arr[i] ==arr[i+m]:
                counter +=1
            else:
                # 一旦当前数字不能作为重复的数字，那么之前的记录也都不能用了
                counter =0
        
            if counter == m*(k-1):
                return True
        return False

            for(int j =0; j< cols; j++)
                if(mat[i][j])
                {
                    f[i][j] =1;
                    if(i) f[i][j] += f[i-1][j];
                }
                else
                    f[i][j] =0;
        
        int res =0;
        for(int i =0; i < rows; i++)
        {
            stack<pair<int, int>> stk;
            for(int j =0; j< cols; j++)
            {
                int s =0;
                
                while(stk.size() && f[i][stk.top().first] >= f[i][j]) stk.pop();
                
                if(stk.size())
                {
                    s += stk.top().second;
                    s += f[i][j] *(j - stk.top().first);
                }
                else
                    s += f[i][j] *(j+1);
                stk.push({j, s});
                res += s;
                    
            }
        }
        return res;
    }

class Solution:
    def partition(self,arr, l, r):
        if l >r: return
        key = arr[r]
        while l <r:
            while l <r and arr[l] >= key: l +=1
            arr[r] = arr[l]
            while l < r and arr[r] <= key: r -=1
            arr[l] = arr[r]
        arr[r] =key
        return l
    
    def findKthLargest(self, nums: List[int], k: int) -> int:
        if not  nums: return 
        l, r = 0, len(nums) -1
        while l <r:
            ind = self.partition(nums, l, r)
            if ind < k-1:
                l  =ind +1
            elif ind ==k-1:
                break
            else:
                r =ind -1
        return nums[k-1]

                r += 1

                while l <r and dif_cnt >i:
                    ind = ord(s[l]) - ord('a')
                    if times[ind] ==1:
                        dif_cnt -= 1
                    if times[ind] ==k:
                        cnt -=1
                    times[ind] -=1
                    l +=1
                    
                if i ==dif_cnt ==cnt:
                    res =max(res, r -l)
        return res
